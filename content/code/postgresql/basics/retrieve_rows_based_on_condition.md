## Create Table

{{< highlight sql >}}
-- Create table called adventurers
CREATE TABLE adventurers (
    -- string variable
    name varchar(255),
    -- integer variable
    age int,
    -- string variable
    race varchar(255),
    -- string variable
    weapon varchar(255)
)
{{< /highlight >}}

## Insert Rows

{{< highlight sql >}}
-- Insert into the table adventurers
INSERT INTO adventurers (name, age, race, weapon)
VALUES ('Fjoak Doom-Wife', 28, 'Human', 'Axe'),
       ('Alooneric Cortte', 29, 'Elf', 'Bow'),
       ('Piperel Ramsay', 35, 'Elf', 'Sword'),
       ('Casimir Yardley', 14, 'Elf', 'Magic')
{{< /highlight >}}

## Retrieve Rows Of Elves

{{< highlight sql >}}
-- Retrieve all rows from table
SELECT * FROM adventurers
    -- Where the race is Elf
    WHERE race = 'Elf'

## Retrieve Rows Of Non-Elves

{{< highlight sql >}}
-- Retrieve all rows from table
SELECT * FROM adventurers
    -- Where the race is not Elf
    WHERE race <> 'Elf'

## Retrieve Rows Where Age Is Over 30

{{< highlight sql >}}
-- Retrieve all rows from table
SELECT * FROM adventurers
    -- Where the age is over 30
    WHERE age > 30

## Retrieve Rows Where Age Is Greater Than Or Equal To 20

{{< highlight sql >}}
-- Retrieve all rows from table

    -- Where the age is greater than or equal to 20
    WHERE age >= 20
